**Component requirements:**
- Client component ("use client") for clipboard interaction
- Display email address prominently

**Implementation:**
```tsx
"use client";

import { useState } from "react";
import { Copy, Check } from "lucide-react";

export function CopyableEmail({ email }: CopyableEmailProps) {
  const [copied, setCopied] = useState(false);

  const handleCopy = async () => {
    try {
      await navigator.clipboard.writeText(email);
      setCopied(true);
      setTimeout(() => setCopied(false), 2000); // Reset after 2 seconds
    } catch (err) {
      console.error("Failed to copy:", err);
    }
  };

  return (
    <button
      onClick={handleCopy}
      className="group flex items-center gap-3 ..."
      aria-label={copied ? "Email copied" : "Copy email to clipboard"}
    >
      <span className="text-xl md:text-2xl font-mono">{email}</span>
      {copied ? (
        <Check className="w-5 h-5 text-green-500" />
      ) : (
        <Copy className="w-5 h-5 opacity-50 group-hover:opacity-100" />
      )}
    </button>
  );
}
```

**Styling:**
- Email text: large, monospace for clarity (text-xl md:text-2xl font-mono)
- Color: --color-text
- Copy icon: --color-primary on hover
- Hover state: subtle background highlight
- Success state: checkmark icon, brief green color
- Cursor: pointer on hover
- Consider adding "Click to copy" tooltip or hint text
